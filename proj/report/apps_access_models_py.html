42 def has_access(self, obj, add_attr=True): 

43 r""" 

44 Determine if the user has frontend update access to change a model or model instance.  

45 Returns True if the user has update access, False otherwise. If a model instance is  

46 provided, the instance is modified with a `user_has_access` attribute that reports the same  

47 True or False value. Per instance access is supported if the model defines `Group` field  

48 like so: 

49 

50 ```python 

51 from django.contrib.auth.models import Group 

52 

53 class Page(models.Model): 

54 groups = models.ManyToManyField(Group, blank=True) 

55 ``` 

56 

57 The algorithm works like so: 

58 

59 - If the user is superuser, access is automatically granted. 

60 - If the user is not a staff member access is automatically rejected. 

61 - Only if the user is a staff member is access checked.  

62 

63 Staff member access rules vary depending on if the `obj` is an model instance or model  

64 class. If `obj` is a model instance: 

65 

66 1. Check that the user has admin `change` permission for the model class via the standard  

67 Django permissions API. If yes, grant access. If no, continue to 2. 

68 2. If instance does not have `groups` attribute, reject access. Else, go to 3. 

69 3. See if user and instance share any groups. If so, grant access Else, reject. 

70 

71 If `obj` is a model class: 

72 

73 1. Check that the user has admin `change` permission for the model class via the standard  

74 Django permissions API. If yes, grant access. If no, reject. 

75 

76 Using this code, you could define a `Page` instance that only select staff members can  

77 access. To check for user access on a model instance, do something like this: 

78  

79 ```python 

80 page = Page.object.get(pk=14) 

81 if user.has_access(page): 

82 print("Access to page granted!")  

83 ``` 

84  

85 In the above example, `page` is monkey patched to now include the `user_has_access`  

86 attribute. So after calling `user.has_access(page)` this works as well: 

87  

88 ```python 

89 if page.user_has_access: 

90 print("Access to page granted!") 

91 ``` 

92 

93 You can override this option (such that the `user_has_access` attribute is not monkey  

94 patched to the instance) by setting the `add_attr` argument to False. It is defaulted to  

95 True. 

96 

97 The object passed in need not be a model instance. It can be a model, e.g. `Page`: 

98 

99 ```python 

100 if user.has_access(Page): 

101 print("Access to Page model granted!")  

102 ``` 

103 

104 But because the object is a model, no `has_access` attribute can be, or will be, affixed to  

105 the model i.e. `Page.user_has_access` will return an error. After all, you can't monkey 

106 patch a class, just an instance of a class. 

107 

108 """ 

109 

110 perm = None 

111 inst = None 

112 is_instance = False 

113 

114 # Is obj a model or a model instance? Don't know of an official way to do this in Django. 

115 # I found that `type(obj.id)` is `int` if a model instance. (If a model, `type(obj.id)` 

116 # returns `django.db.models.query_utils.DeferredAttribute`. Just FYI.) 

117 is_instance = type(obj.id) == int 

118 if is_instance: inst = obj 

119 

120 # Get the change model permission name (e.g. 'master.change_post'). 

121 perm = f'{obj._meta.app_label}.change_{obj._meta.verbose_name}' 

122 

123 # If user is not active, reject access request. 

124 if not self.is_active: 

125 if is_instance and add_attr: setattr(inst, 'user_has_access', False) 

126 return False 

127 

128 # If user is a superuser, grant access request. (Superusers don't need to have group  

129 # affiliation to modify.) 

130 if self.is_superuser: 

131 if is_instance and add_attr: setattr(inst, 'user_has_access', True) 

132 return True 

133 

134 # If user is not staff, reject access request. 

135 if not self.is_staff: 

136 if is_instance and add_attr: setattr(inst, 'user_has_access', False) 

137 return False 

138 

139 # Value `perm` is the standard Django model change permissions (e.g. 'master.change_page').  

140 # Check to see if user has Django model change permission. If so grant access request. 

141 if self.has_perm(perm): 

142 if is_instance and add_attr: setattr(inst, 'user_has_access', True) 

143 return True 

144 # If obj is a model then reject access request because self.has_perm(perm) from above 

145 # returned False. Note, we don't reject access if obj is a model instance as we have  

146 # not yet checked instance access.  

147 else: 

148 if not is_instance: 

149 return False 

150 

151 # If inst is defined ... 

152 if is_instance: 

153 # If the `groups` field is not defined for the model, reject access.  

154 if not hasattr(inst, 'groups'): 

155 if add_attr: setattr(inst, 'user_has_access', False) 

156 return False 

157 # Check to see if user group set and the instance group set have at least one common 

158 # group. If so, grant access. Else, request access. 

159 user_has_inst_group = any(group in inst.groups.all() for group in self.groups.all()) 

160 if add_attr: setattr(inst, 'user_has_access', user_has_inst_group) 

161 return user_has_inst_group 

162 # Should be impossible get here, but more explicit if included.  

163 else: # pragma: no cover 

164 return False # pragma: no cover 

165 

166 # Should be impossible get here, but more explicit if included. 

167 return False # pragma: no cover
