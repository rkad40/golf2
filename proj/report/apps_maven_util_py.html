51def get_user_access(user, folder): 

52 r""" 

53 Get access level: 

54 

55 0. Restricted - no access; can see folder tree but not files 

56 1. Visitor - can browse files 

57 2. Contributor - can create files and edit them 

58 3. Editor - can edit files created by any user 

59 4. Admin - can do anything 

60 """ 

61 # If user is superuser, policy access level is 4 (highest access level). 

62 if user is not None and user.is_superuser: return 4 

63 # Find the folder policy. If the current folder does not have a policy, search ancestors  

64 # beginning with the most recent. When a folder policy is found, terminate search and use it. 

65 policy = folder.policy 

66 if policy is None: 

67 for ancestor_folder in folder.ancestor_folders.all().order_by('-url'): 

68 # if DEBUG: print(ancestor_folder.url) 

69 policy = ancestor_folder.policy 

70 if policy is not None: break 

71 # If no policy is found, superusers have level 4 access and all other users have level 1  

72 # access, even visitors. 

73 if policy is None: 

74 if user is None or user.is_anonymous: return 1 

75 if user.is_superuser: return 4 

76 return 1 

77 # You can only get here if a folder policy was found. If user is None, user level is 1 if  

78 # the policy allows visitors to view, 0 otherwise. 

79 if settings.DEBUG: 

80 if DEBUG: print(f'User: {user}') 

81 if DEBUG: print(f'Folder policy: {policy.name}') 

82 if user is None or user.is_anonymous: 

83 if policy.visitors_can_view: return 1 

84 else: return 0 

85 # Compare user groups agains folder policy admin, editor and contrib groups returning the first 

86 # level match if one is made. 

87 if any(group in policy.admin_groups.all() for group in user.groups.all()): return 4 

88 if any(group in policy.editor_groups.all() for group in user.groups.all()): return 3 

89 if any(group in policy.contrib_groups.all() for group in user.groups.all()): return 2 

90 # If no user group matches an admin, editor or contrib group, level is 1. 

91 return 1
